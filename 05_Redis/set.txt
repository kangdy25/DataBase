# song:1 투표 세트에 유저 추가
SADD votes:song:1 user:1
SADD votes:song:1 user:2

# song:1에 투표한 모든 유저 조회
SMEMBERS votes:song:1

# 특정 유저가 song:1에 투표했는지 확인
SISMEMBER votes:song:1 user:1  # 있음 → 1
SISMEMBER votes:song:1 user:3  # 없음 → 0

# song:1 투표자 수 확인
SCARD votes:song:1

# song:2 투표 세트에 여러 유저 추가
SADD votes:song:2 user:3 user:4 user:5

# song:2에 투표한 모든 유저 조회
SMEMBERS votes:song:2

# 교집합(두 곡 모두 투표한 유저)
SINTER votes:song:1 votes:song:2

# user:3을 song:1에도 추가 (교집합 생김)
SADD votes:song:1 user:3
SINTER votes:song:1 votes:song:2  # 이제 user:3 포함됨

# 차집합(song:1에는 있지만 song:2에는 없는 유저)
SDIFF votes:song:1 votes:song:2

# 합집합(두 곡 중 하나라도 투표한 유저)
SUNION votes:song:1 votes:song:2

# 특정 유저 제거
SREM votes:song:1 user:1
